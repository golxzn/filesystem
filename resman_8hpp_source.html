13template<class T>
14struct transparent_hash {
15 using hash_type = std::hash<T>;
16 using is_transparent = void;
17
18 std::size_t operator()(const T &v) const noexcept { return hash_type{}(v); }
19};

21template<>
22struct transparent_hash<std::wstring> {
23 using char_type = std::wstring::value_type;
24 using hash_type = std::hash<std::wstring_view>;
25 using is_transparent = void;
26
27 std::size_t operator()(const std::wstring &v) const noexcept { return hash_type{}(v); }
28 std::size_t operator()(const std::wstring_view &v) const noexcept { return hash_type{}(v); }
29 std::size_t operator()(const char_type *const v) const noexcept { return hash_type{}(v); }

29template<class T>
30struct transparent_hash {
31 using hash_type = std::hash<T>;
32 using is_transparent = void;
33
34 size operator()(const T &v) const noexcept { return hash_type{}(v); }
35};
36
37template<>
38struct transparent_hash<std::wstring> {
39 using char_type = std::wstring::value_type;
40 using hash_type = std::hash<std::wstring_view>;
41 using is_transparent = void;
42
43 size operator()(const std::wstring &v) const noexcept { return hash_type{}(v); }
44 size operator()(const std::wstring_view &v) const noexcept { return hash_type{}(v); }
45 size operator()(const char_type *const v) const noexcept { return hash_type{}(v); }
46};
